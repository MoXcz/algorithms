Insertion Sort

        def insertionSort(arr):
            n = len(arr) 
            for j in range(1, n):  
            key = arr[j] 
            i = j-1
            while i >= 0 and arr[i] > key: 
                arr[i+1] = arr[i]
                i -= 1
            arr[i+1] = key 

        arr = [12, 11, 13, 5, 6]
        insertionSort(arr)
        print(arr)
