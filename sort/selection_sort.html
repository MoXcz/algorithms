Selection Sort

    def selectionSort(array, n):
        for index in range(n):
            min_index = index
            for j in range(index + 1, n):
                if array[j] < array[min_index]:
                    min_index = j
            (array[index], array[min_index]) = (array[min_index], array[index])
        
    arr = [12, 11, 13, 10, 5, 6, 1]
    n = len(arr)
    selectionSort(arr, n)
    print(arr)
